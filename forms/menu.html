const API_URL = 'http://localhost:3000'; async function carregarDashboard() { const token = localStorage.getItem('token'); if (!token) return window.location.href = 'index.html'; // Saudação const name = localStorage.getItem('name'); if (name) { document.getElementById('userGreeting').textContent = `👋 ${name}`; } try { // Valor Total dos Objetos const res = await fetch(`${API_URL}/objetos/valorTotal`, { headers: { 'Authorization': `Bearer ${token}` } }); if (res.ok) { const data = await res.json(); const valor = Number(data.valortotal || 0); document.getElementById('valorTotal').textContent = valor.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' }); } else { document.getElementById('valorTotal').textContent = 'R$ 0,00'; } } catch (error) { console.error('Erro ao carregar valor total:', error); document.getElementById('valorTotal').textContent = 'R$ 0,00'; } } window.onload = carregarDashboard; function logout() { localStorage.clear(); window.location.href = 'index.html'; }
